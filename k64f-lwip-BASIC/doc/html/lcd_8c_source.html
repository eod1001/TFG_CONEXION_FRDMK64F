64 void LCD_Init(uint8_t lcd_addr, uint8_t lcd_cols, uint8_t lcd_rows, uint8_t charsize)
65 {
66  g_lcd_addr = lcd_addr;
67  g_lcd_cols = lcd_cols;
68  g_lcd_rows = lcd_rows;
69  g_lcd_charsize = charsize;
70  g_lcd_backlightval = LCD_BACKLIGHT;
71 
72  select_slave(g_lcd_addr);
73 
74  /* Set bit mode, number of rows, and font size. */
75  g_lcd_displayfunction = LCD_4BITMODE | LCD_2LINE | LCD_5x8DOTS;
76 
77  /* For some 1 line displays you can select a 10 pixel high font. */
78  if ((g_lcd_charsize != 0) && (g_lcd_rows == 1))
79  {
80  g_lcd_displayfunction |= LCD_5x10DOTS;
81  }
82 
83  /* Set 4 bit mode. */
84  LCD_write4bits(0x03 << 4);
85  /* New attempt, just in case.*/
86  LCD_write4bits(0x03 << 4);
87  /* Set 4 bit interface. */
88  LCD_write4bits(0x02 << 4);
89 
90  /* Issue the setting command. */
91  LCD_command(LCD_FUNCTIONSET | g_lcd_displayfunction);
92 
93  /* Set the display on, no cursor, nor blinking by default. */
94  g_lcd_displaycontrol = LCD_DISPLAYON | LCD_CURSOROFF | LCD_BLINKOFF;
95  LCD_command(LCD_ENTRYMODESET | g_lcd_displayfunction);
96 
97  /* Set the text direction. */
98  g_lcd_displaymode = LCD_ENTRYLEFT | LCD_ENTRYSHIFTDECREMENT;
99 
100  /* Go home. */
101  LCD_home();
102 }
