256  netconn_listen(netconn);
257

260  LCD_clear_row(ROW_BOTTOM);
261  LCD_printstr(port_announce);
262 
263  for (;;)
264  {
265  netconn_accept(netconn, &newconn);
266 
267  while (ERR_OK == netconn_recv(newconn, &netbuf))
268  {
269  do
270  {
271  netbuf_copy(netbuf, buffer, sizeof(buffer));
272  buffer[netbuf->p->tot_len] = '\0';
273 
274  rx_command_check(buffer, netbuf->p->tot_len);
275  }
276  while (netbuf_next(netbuf) >= 0);
277 
278  netbuf_delete(netbuf);
279  }
280 
281  netconn_delete(newconn);
282  }
